template class sophus::CameraModelT¶

Overview¶

Camera model class template for pinhole-like camera projections. More…

#include <camera_model.h>

template <class ScalarT, class ProjectionT>
class CameraModelT {
public:
    // typedefs

    using Proj = ProjectionT ;
    using PointCamera = Eigen::Matrix<ScalarT, 3, 1> ;
    using PixelImage = Eigen::Matrix<ScalarT, 2, 1> ;
    using ProjInCameraZ1Plane = Eigen::Matrix<ScalarT, 2, 1> ;
    using Params = Eigen::Matrix<ScalarT, kNumParams, 1> ;
    using DistorationParams = Eigen::Matrix<ScalarT, kNumDistortionParams, 1> ;

    // fields

    static constexpr int kNumDistortionParams = Proj::kNumDistortionParams;
    static constexpr int kNumParams = Proj::kNumParams;
    static const constexpr std::string_view kProjectionModel =       Proj::kProjectionModel;

    // construction

    CameraModelT(const ImageSize& image_size, const Params& params);

    // methods

    ScalarT fx() const;
    ScalarT fy() const;
    ScalarT cx() const;
    ScalarT cy() const;
    DistorationParams distortionParams() const;
    Eigen::Matrix<ScalarT, kNumParams, 1>& mutParams();
    const Eigen::Matrix<ScalarT, kNumParams, 1>& params() const;
    CameraModelT subsampleDown() const;
    CameraModelT subsampleUp() const;
    CameraModelT binDown() const;
    CameraModelT binUp() const;
    CameraModelT scale(const ImageSize& image_size) const;
    CameraModelT roi(const Eigen::Vector2i& top_left, ImageSize roi_size) const;
    PixelImage warp(const ProjInCameraZ1Plane& point2_in_camera_z1_plane) const;
    PixelImage camProj(const PointCamera& point_in_camera) const;
    Eigen::Matrix<ScalarT, 2, 3> dxCamProjX(const PointCamera& point_in_camera) const;
    ProjInCameraZ1Plane unwarp(const PixelImage& pixel_in_image) const;
    Eigen::Matrix<ScalarT, 2, 2> dxWarp(const PixelImage& pixel_in_image) const;
    PointCamera camUnproj(const PixelImage& pixel_in_image, double depth_z) const;
    ScalarT* mutData();
    const ImageSize& imageSize() const;
    bool contains(const Eigen::Vector2i& obs, int border = 0) const;
    bool contains(const PixelImage& obs, ScalarT border = ScalarT(0)) const;
    static CameraModelT fromData(ScalarT const*const ptr);
};

Subsamples pixel down, factor of 0.5.
