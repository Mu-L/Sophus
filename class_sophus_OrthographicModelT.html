Overview¶

Orthographic camera model class template. More…

#include <camera_model.h>

template <class ScalarT>
class OrthographicModelT {
public:
    // typedefs

    using Proj = OrthographicProjection ;
    using PointCamera = Eigen::Matrix<ScalarT, 3, 1> ;
    using PixelImage = Eigen::Matrix<ScalarT, 2, 1> ;
    using Params = Eigen::Matrix<ScalarT, kNumParams, 1> ;
    using DistorationParams = Eigen::Matrix<ScalarT, kNumDistortionParams, 1> ;

    // fields

    static constexpr int kNumDistortionParams = Proj::kNumDistortionParams;
    static constexpr int kNumParams = Proj::kNumParams;
    static const constexpr std::string_view kProjectionModel =       Proj::kProjectionModel;

    // construction

    OrthographicModelT(
        ImageSize image_size,
        const Params& params
        );

    OrthographicModelT(
        ImageSize image_size,
        const Eigen::Matrix<ScalarT, 2, 1>& scale,
        const Eigen::Matrix<ScalarT, 2, 1>& offset
        );

    // methods

    ScalarT scaleX() const;
    ScalarT scaleY() const;
    ScalarT offsetX() const;
    ScalarT offsetY() const;
    ScalarT left() const;
    ScalarT right() const;
    ScalarT top() const;
    ScalarT bottom() const;
    OrthographicModelT subsampleDown() const;
    OrthographicModelT subsampleUp() const;
    OrthographicModelT binDown() const;
    OrthographicModelT binUp() const;
    DistorationParams distortionParams() const;
    Eigen::Matrix<ScalarT, kNumParams, 1>& mutParams();
    const Eigen::Matrix<ScalarT, kNumParams, 1>& params() const;
    PixelImage camProj(const PointCamera& point_camera) const;

    PointCamera camUnproj(
        const PixelImage& pixel_image,
        double depth_z
        ) const;

    ScalarT* mutData();
    const ImageSize& imageSize() const;
    bool contains(const Eigen::Vector2i& obs, int border = 0) const;
    bool contains(const PixelImage& obs, ScalarT border = ScalarT(0)) const;
    static OrthographicModelT fromData(ScalarT const*const ptr);
};
